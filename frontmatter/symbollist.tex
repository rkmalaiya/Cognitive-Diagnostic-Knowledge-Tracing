\symbolentry{$f: D \rightarrow R$}{Function on domain D with range R}
\symbolentry{$a$ or $A$}{Scalar variable}
\symbolentry{${\bf a}$}{Column vector (lowercase boldface)}
\symbolentry{${\bf a}^T$}{Transpose of column vector {\bf a}}
\symbolentry{${\bf A}$}{Matrix (uppercase boldface)}
\symbolentry{$\tilde{a}$}{Scalar random variable}
\symbolentry{${\bf \tilde{a}}$}{Random column vector}
\symbolentry{${\bf \tilde{A}}$}{Random matrix}
\symbolentry{$\hat{a}$}{Random scalar-valued function}
\symbolentry{${\bf \hat{a}}$}{Random vector-valued function}
\symbolentry{${\bf \hat{A}}$}{Random matrix-valued function}
\symbolentry{$d{\bf f}({\bf x}_0)/d{\bf x}$}{Jacobian of function ${\bf f}$ evaluated at ${\bf x}_0$}
\symbolentry{$\exp(x)$}{Exponential function evaluated at $x$}
\symbolentry{${\bf exp}({\bf x})$}{Vector whose $i$th element, $x_i$, is $\exp(x_i)$}
\symbolentry{$\log(x)$}{Natural logarithm function evaluated at $x$}
\symbolentry{${\bf log}({\bf x})$}{Vector whose $i$th element, $x_i$, is $\log(x_i)$}
\symbolentry{${\cal S}(x)$}{Logistic sigmoid ${\cal S}(x) = 1/(1+\exp(-x))$}
\symbolentry{${\cal J}(x)$}{Softplus sigmoid ${\cal J}(x)=\log(1 + \exp(x))$}
\symbolentry{$|{\bf x}|$}{Square root of ${\bf x}^T{\bf x}$}
\symbolentry{${\bf I}_d$}{Identity matrix with dimension $d$}
\symbolentry{$\det({\bf M})$}{Determinant of matrix ${\bf M}$}
\symbolentry{$tr({\bf M})$}{Trace of matrix ${\bf M}$}
\symbolentry{${\bf 1}_d$}{Column vector of ones with dimension $d$}
\symbolentry{${\bf 0}_d$}{Column vector of zeros with dimension $d$}
\symbolentry{$A \times B$}{Cartesian product of set A and set B}
\symbolentry{${\cal R}^d$}{Set of $d$-dimensional real column vectors}
\symbolentry{${\cal R}^{m \times n}$}{Set of $m$ by $n$ real matrices}
\symbolentry{${\bfs \theta}$}{Parameters of a learning machine}
\symbolentry{${\cal D}_n$}{Data sample of $n$ pattern vectors}
\symbolentry{$\hat{\ell}_n$}{Empirical risk function for ${\cal D}_n$}
\symbolentry{$\ell$}{The risk function estimated by $\hat{\ell}_n$}
\symbolentry{$p_e$}{Environmental probability density}
\symbolentry{$p({\bf x})$}{Probability density for ${\bf \tilde{x}}$}
\symbolentry{$p({\bf x};{\bfs \theta})$}{Density of ${\bf \tilde{x}}$ with parameter ${\bfs \theta}$}
\symbolentry{$p({\bf x}|{\bfs \theta})$}{Density of ${\bf \tilde{x}}$ given ${\bfs \theta}$}
\symbolentry{$E\{{\bf f}({\bf \tilde{x}})\}$}{Expectation of $f({\bf x})$ with respect to density of ${\bf \tilde{x}}$}
\symbolentry{${\cal M}$}{Probability model}
\symbolentry{${\cal H}({\bf \tilde{x}})$ or ${\cal H}(p)$}{Entropy of random vector ${\bf \tilde{x}}$ with density $p$}
\symbolentry{${\cal H}(p_e, p)$}{Cross entropy of density $p$ with respect to $p_e$}
\symbolentry{$D(p_e||p)$}{Kullback-Leibler divergence of density $p$ from $p_e$}
